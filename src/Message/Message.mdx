import { Playground, Props } from 'docz' 
import { createMuiTheme, ThemeProvider } from '@material-ui/core/styles';
import Button  from '../Button';
import Icon  from '../Icon'
import message  from './';
import { setTheme } from '../theme'


# message

## Props

<Props of={message} />

## Basic usage 

<Playground>
  {
    () => {
      const theme = createMuiTheme({
        palette: {
          primary: { main: 'rgba(35, 178, 190, 1)', contrastText: '#fff' },
          error: { main: 'rgba(255, 152, 0, 1)' },
          text: { disabled: 'rgba(0, 0, 0, 0.26)' },
          action: { hover: 'rgba(0, 0, 0, 0.04)' },
        },
      });
      setTheme(theme)
      const showMessage = () => {
        message.open({
          noticeIconName: 'notifications',
          content: '转诊成功！',
          style: { left: '50%', bottom: '20px', transform: 'translateX(-50%)', },
          expandActions: [{
            text: '查看患者',
            iconName: 'visibility',
            onClick: () => console.log('successful'),
          }]
        })
      }
      const showMsg = () => {
        message.open({
          content: '密码设置成功！',
          style: { left: '50%', bottom: '20px', transform: 'translateX(-50%)', }
        })
      }
      return (
        <ThemeProvider theme={theme}>
          <div>
            <Button variant="contained" onClick={showMessage}>显示Message</Button>
            <br/><br/>
            <Button variant="contained" onClick={showMsg}>显示Message2</Button>
          </div>
        </ThemeProvider>
      )
    }
  }
</Playground>
