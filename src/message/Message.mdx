## Basic usage 

<Playground>
  {
    () => {
      const theme = createMuiTheme({
        palette: {
          primary: { main: 'rgba(35, 178, 190, 1)', contrastText: '#fff', light: '#18FFFF' },
        },
      });
      setTheme(theme)
      const showMsg = () => {
        message.open({
          isSingleton: false,
          type: 'primary',
          noticeIconName: 'notification',
          content: '转诊成功！',
          style: { left: '50%', bottom: '20px', transform: 'translateX(-50%)' },
          expandActions: [{
            text: '查看患者',
            iconName: 'notification',
            onClick: () => console.log('successful'),
          }],
          duration: 30,
        })
      }
      const showMsg2 = () => {
        message.open({
          type: 'warning',
          noticeIconName: 'notification',
          content: '密码设置成功！',
          style: { left: '50%', bottom: '20px', transform: 'translateX(-50%)', }
        })
      }
      const showMsg3 = () => {
        message.open({
          content: '密码设置成功！',
          style: { right: '50px', bottom: '20px', }
        })
      }
      const destroyMessage = () => {
      message.destroy()
      }
      return (
        <ThemeProvider theme={theme}>
          <div>
            <Button variant="contained" onClick={showMsg}>显示Message</Button>
            <br/><br/>
            <Button variant="contained" onClick={showMsg2}>显示Message2</Button>
            <br/><br/>
            <Button variant="contained" onClick={showMsg3}>显示Message3</Button>
            <br/><br/>
            <Button variant="contained" onClick={destroyMessage}>销毁message</Button>
          </div>
        </ThemeProvider>
      )
    }
  }
</Playground>
