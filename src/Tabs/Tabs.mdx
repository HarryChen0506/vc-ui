## Basic usage 

<Playground>
  {() => {
    const [curTab, changeTab] = React.useState('project');
    const handleTabChange = (val, e) => {
        changeTab(val)
    };
    return (
        <Tabs value={curTab} onChange={handleTabChange}>
            <Tab label="项目" value="project" /> 
            <Tab label="用户" value="user" disabled /> 
            <Tab label="机构" value="orgnization" /> 
         </Tabs>
    );
  }}
</Playground>
