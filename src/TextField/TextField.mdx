## Basic usage 

<Playground>
  {() => {
    const [state, setState] = React.useState({ text: '', number: undefined, password: '', select: 10 })
    const handleChange = (event, key) => {
      setState({ ...state, [key]: event.target.value });
    };
    return (
      <div>
        <TextField label='普通输入' type='text' value={state.text} error helperText='必填' onChange={e => handleChange(e, 'text')} />
        <br/>
        <TextField label='密码输入' required type='password' value={state.password} onChange={e => handleChange(e, 'password')} />
        <br/>
        <TextField label='数字输入' type='number' value={state.number} onChange={e => handleChange(e, 'number')} />
        <br/>
        <TextField label='下拉选择' type='select' value={state.select} onChange={e => handleChange(e, 'select')}>
          <MenuItem value={10}>Ten</MenuItem>
          <MenuItem value={20}>Twenty</MenuItem>
          <MenuItem value={30}>Thirty</MenuItem>
        </TextField>
      </div>
    )
  }}
</Playground>
